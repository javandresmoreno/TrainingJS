                      if (!texto) return console.warn("No ingresaste un texto válido");

                    const textoEnCadena = (cadena = "", texto = "") => {
                      if (!cadena) return console.warn("No ingresaste el texto largo.");
                      if (!texto) return console.warn("No ingresaste la palabra a evaluar");

                      let i = 0,
                        contador = 0;
                     
                      while (i !== -1) {
                     
                        i = cadena.indexOf(texto, i);

                        if (i !== -1) {
                          i++;
                          contador++;
                        }
                      }
                      return console.info(`La palabra ${texto} se repite ${contador} veces`);
                    };

                    textoEnCadena(); // No ingresaste el texto largo
                    textoEnCadena("", "mundo"); // No ingresaste el texto largo
                    textoEnCadena("hola mundo adios mundo"); // No ingresaste la palabra a evaluar
                    textoEnCadena("hola mundo adios mundo hola adios mundo", "mundo"); //La palabra mundo se repite 3 veces

                    console.groupEnd();

    Mi Solución

    Solución del Profe Jon
    
                  
                    /*
                      Primero realizamos las validaciones:
                      - Si no hay palabra, lanzamos un warning.
                      - Convertimos nuestra palabra a minúsculas, porque JS es sensitivo a letras 
                        mayúsculas y minúsculas. 
                      - Defino la variable alReves por medio de los métodos split, para volverlo un array,
                        reverse para darle la vuelta al array y join para unir de nuevo en una cadena de 
                        texto. 
                      - Retorno que si la palabra es estrictamente igual a alReves entonces me imprima 
                        según sea el caso 
                    */

                    const palindromo = (palabra = "") => {
                      if (!palabra) return console.warn("No ingresaste una palabra o frase");
                      palabra = palabra.toLowerCase();
                      let alReves = palabra.split("").reverse().join("");

                      return palabra === alReves
                        ? console.info(
                            `SÍ es palíndromo, Palabra original ${palabra}, Palabra al revés ${alReves}`
                          )
                        : console.info(
                            `NO es palíndromo, Palabra original ${palabra}, Palabra al revés ${alReves}`
                          );
                    };

                    palindromo();
                    palindromo("Salas");
                    palindromo("LaTeleLetal");
                    palindromo("Mantenimiento");

                      /*
                        Primero realizamos las validaciones:
                        - Si no hay Texto, lanzamos un warning.
                        - Si no existe patrón, lanza otro warning
                        - si todo está, entonces en el metodo replace agregamos una nueva 
                          instancia del objeto RegExp y pasamos el patrón y las banderas 
                          "ig" que significan que: i: no importa si son mayúsculas o

                          en la primer coincidencia, sino que recorra todo el texto hasta
                          encontrar todas. 
                        
                      */

                      const eliminarCaracteres = (texto = "", patron = "") => {
                        !texto
                          ? console.warn(`No ingresaste texto.`)
                          : !patron
                          ? console.warn(
                              "No ingresaste los caracteres a eliminar (Patrón de caracteres)"
                            )
                          : console.info(texto.replace(new RegExp(patron, "ig"), ""));
                      };

                      eliminarCaracteres();
                      eliminarCaracteres("xyz1, xyz2, xyz3, xyz4, xyz5, xyz6");
                      eliminarCaracteres("xyz1, xyz2, xyz3, xyz4, xyz5, xyz6", "xyz");
                      eliminarCaracteres("xyz1, xyz2, xyz3, xyz4, xyz5, xyz6", "yz");

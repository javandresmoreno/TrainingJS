    MirCha's  Solution
    
              
                //Función declarada
                function contarCaracteres(cadena = "") {
                  if (!cadena) {
                    console.warn("No ingresaste ninguna cadena de texto");
                  } else {
                    console.info(
                      `La cadena de texto "${cadena}" tiene ${cadena.length} caracteres.`
                    );
                  }
                }

                contarCaracteres(); // Output: No ingresaste ninguna cadena de texto
                contarCaracteres("Hola Mundo"); // Output: La Cadena de Texto tiene 10 caracteres 

                //Función Expresada
                const contarCaracteres2 = (cadena = "") => {
                  !cadena
                    ? console.warn("No ingresaste ninguna cadena de texto")
                    : console.info(
                        `La cadena de texto "${cadena}" tiene ${cadena.length} caracteres.`
                      );
                };

                contarCaracteres2();
                contarCaracteres2("Hello World");
